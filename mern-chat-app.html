Core Features

Everything a production
chat needs

Built from scratch during my internship at Navikshaa Technologies — each feature carefully engineered for real users.

⚡
Live Messaging
Bi-directional Socket.IO channels with room-based namespacing. Messages delivered in real-time to all connected clients without polling.
✍️
Typing Indicators
Debounced keydown event listeners emit typing events via socket. Auto-clear timeout prevents stale indicators.
✓✓
Read Receipts
Message delivery state tracked in MongoDB. Acknowledgement events update sender UI when recipient opens the conversation.
🔐
JWT Authentication
Stateless auth with access + refresh token rotation. Tokens verified on each HTTP request and Socket.IO handshake.
🛡️
Role-Based Access Control
Middleware-enforced RBAC with admin, moderator, and user tiers. Room creation, deletion, and moderation gated by role.
📱
Responsive UI
React.js frontend with code-split bundles, lazy-loaded routes, and Tailwind breakpoints. Optimized for mobile, tablet, and desktop.

// socket-server.js — Socket.IO event handlers
const io = require('socket.io')(server, {
  cors: { origin: process.env.CLIENT_URL, credentials: true }
});

// Auth middleware for socket handshake
io.use(async (socket, next) => {
  const token = socket.handshake.auth.token;
  try {
    socket.user = jwt.verify(token, process.env.JWT_SECRET);
    next();
  } catch (err) { next(new Error('Unauthorized')); }
});

io.on('connection', (socket) => {
  // Join room on connect
  socket.on('join:room', (roomId) => {
    socket.join(roomId);
    socket.to(roomId).emit('user:joined', { userId: socket.user._id });
  });

  // Handle incoming message
  socket.on('message:send', async ({ roomId, content }) => {
    const msg = await Message.create({
      sender: socket.user._id, room: roomId,
      content, status: 'delivered'
    });
    io.to(roomId).emit('message:new', msg);
  });

  // Typing indicator with debounce
  socket.on('typing:start', ({ roomId }) => {
    socket.to(roomId).emit('typing:update', {
      userId: socket.user._id, isTyping: true
    });
  });

  // Read receipt acknowledgement
  socket.on('message:read', async ({ messageId, roomId }) => {
    await Message.findByIdAndUpdate(messageId, { status: 'read' });
    io.to(roomId).emit('message:status', { messageId, status: 'read' });
  });
});
// auth.middleware.js — JWT verification
const jwt = require('jsonwebtoken');
const { TokenBlacklist } = require('../models');

const authenticate = async (req, res, next) => {
  const authHeader = req.headers['authorization'];
  if (!authHeader?.startsWith('Bearer '))
    return res.status(401).json({ error: 'No token provided' });

  const token = authHeader.split(' ')[1];
  const isBlacklisted = await TokenBlacklist.findOne({ token });
  if (isBlacklisted)
    return res.status(401).json({ error: 'Token revoked' });

  try {
    req.user = jwt.verify(token, process.env.JWT_SECRET);
    next();
  } catch (err) {
    if (err.name === 'TokenExpiredError')
      return res.status(401).json({ error: 'Token expired' });
    res.status(403).json({ error: 'Invalid token' });
  }
};

module.exports = { authenticate };
// rbac.middleware.js — Role-based access control
const PERMISSIONS = {
  admin:     ['read', 'write', 'delete', 'manage'],
  moderator: ['read', 'write', 'delete'],
  user:      ['read', 'write'],
};

const authorize = (...requiredPerms) => (req, res, next) => {
  const userRole = req.user?.role;
  const userPerms = PERMISSIONS[userRole] || [];
  const hasAll = requiredPerms.every(p => userPerms.includes(p));

  if (!hasAll) {
    return res.status(403).json({
      error: `Requires permissions: ${requiredPerms.join(', ')}`
    });
  }
  next();
};

// Usage: router.delete('/room/:id', authenticate, authorize('delete'), deleteRoom)

// useSocket.js — Custom React hook for Socket.IO
import { useEffect, useRef, useCallback } from 'react';
import { io } from 'socket.io-client';
import { useAuthStore } from '../store/auth';

export const useSocket = (roomId) => {
  const socketRef = useRef(null);
  const typingTimer = useRef(null);
  const token = useAuthStore(s => s.token);

  useEffect(() => {
    socketRef.current = io(process.env.REACT_APP_API_URL, {
      auth: { token }, transports: ['websocket']
    });

    socketRef.current.emit('join:room', roomId);
    return () => socketRef.current.disconnect();
  }, [roomId, token]);

  const sendMessage = useCallback((content) => {
    socketRef.current?.emit('message:send', { roomId, content });
  }, [roomId]);

  const emitTyping = useCallback(() => {
    socketRef.current?.emit('typing:start', { roomId });
    clearTimeout(typingTimer.current);
    typingTimer.current = setTimeout(() =>
      socketRef.current?.emit('typing:stop', { roomId }), 2000
    );
  }, [roomId]);

  return { sendMessage, emitTyping, socket: socketRef.current };
};
